        <button onclick="askAI()">Send</button>

    <script>
        let questionCount = 0;

        function askAI() {
            const input = document.getElementById('userInput');
            const messages = document.getElementById('messages');
            
            if (input.value.trim() === "") return;

            // Show User Question
            messages.innerHTML += `<div class="user-msg">You: ${input.value}</div>`;
            
            questionCount++;
            let response = "tnak";

            // If asked more than 3 times, change response
            if (questionCount > 3) {
                response = "mtrohch tnik mok?";
            }

            // Show AI Response
            setTimeout(() => {
                messages.innerHTML += `<div class="ai-msg">YCNFR: ${response}</div>`;
                messages.scrollTop = messages.scrollHeight;
            }, 500);

            input.value = "";
        }
    </script>
